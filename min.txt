#!/usr/bin/python3
import sys

def parse_numbers(line):
    return [float(num) for num in line.split()]

numbers = []
for line in sys.stdin:
    numbers.extend(parse_numbers(line.strip()))

if len(numbers) > 0:
    print(min(numbers))
else:
    print("Error: No numbers provided", file=sys.stderr)
    sys.exit(1)
