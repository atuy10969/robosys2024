#!/usr/bin/python3
import sys

def tonum(s):
    try:
        return int(s)
    except:
        return float(s)

# 合計値と入力数を初期化
total = 0
count = 0

for line in sys.stdin:
    line = line.rstrip()
    total += tonum(line)
    count += 1

# 平均値を計算
if count > 0:
    average = total / count
    print(f"平均値: {average}")
else:
    print("入力がありません")
