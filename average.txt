#!/bin/bash

# 標準入力または引数で与えられた数値の平均を計算するスクリプト

# 入力を確認
if [ -p /dev/stdin ]; then
  input=$(cat /dev/stdin)
else
  input="$*"
fi

# 入力が空の場合、エラーを返す
if [ -z "$input" ]; then
  echo "Error: 入力が必要です" >&2
  exit 1
fi

# 入力を処理して平均を計算
sum=0
count=0
for num in $input; do
  if [[ "$num" =~ ^-?[0-9]+(\.[0-9]+)?$ ]]; then
    sum=$(echo "$sum + $num" | bc)
    count=$((count + 1))
  else
    echo "Error: 無効な入力: $num" >&2
    exit 1
fi
done

average=$(echo "$sum / $count" | bc -l)
echo "$average"
